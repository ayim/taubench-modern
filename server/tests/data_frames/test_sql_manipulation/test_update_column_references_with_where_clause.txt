================================================================================
INPUT SQL:
================================================================================
SELECT full_name FROM employees WHERE salary > 50000

================================================================================
MAPPINGS:
================================================================================
{
  "employees": {
    "full_name": "first_name || ' ' || last_name",
    "salary": "annual_salary"
  }
}

================================================================================
OUTPUT SQL:
================================================================================
SELECT
  first_name || ' ' || last_name AS full_name
FROM employees
WHERE
  annual_salary > 50000
