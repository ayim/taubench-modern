WITH customer_agg AS (
  SELECT
    id AS customer_id,
    SUM(total) AS total_spent
  FROM orders
  GROUP BY
    id
)
SELECT
  cm.name AS customer_full_name,
  ca.total_spent
FROM customer_agg AS ca
JOIN customers AS cm
  ON cm.id = ca.customer_id
